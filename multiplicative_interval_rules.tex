\begin{theorem} \label{thm:multiplicative:rules}
  Let $a,b,\epsilon,\delta \in\Rplus$. We have

  \begin{enumerate}
    \item $\ean[\epsilon]{a} \cdot \ean[\delta]{b} = \ean[\epsilon+\delta]{a\cdot b}$
    \item $\ean[\epsilon]{a} = \an[\sinh(\epsilon)a]{\cosh(\epsilon)a}$
  \end{enumerate}
\end{theorem}
