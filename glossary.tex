\newglossaryentry{bigo}{%
	name={\ensuremath{\bigo{\cdot}}},
  sort={O},
	description={Big O notation: $\seq{\epsilon_n} \in \bigo{a_n} \iff \exists C_\infty > 0 : \limsup_{n\to\infty} \abs{\frac{\epsilon_n}{a_n}} \le C_\infty$}

\newglossaryentry{littleo}{%
	name={\ensuremath{\littleo{\cdot}}},

}
\newglossaryentry{phi}{%
	name={\ensuremath{\phi(x)}},
  sort={phi},
	description={The density function of the standardized normal distribution: $\phi(x)=\frac{1}{\sqrt{2\pi}}\e{-\tfrac{x^2}{2}}$}

\newglossaryentry{Phi}{%
	name={\ensuremath{\Phi(x)}},
